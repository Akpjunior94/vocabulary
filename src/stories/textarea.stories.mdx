export const textAreaType = (size) => {
  let sizeClass = size ? ` is-${size}` : ''
  return `
<textarea class="textarea${sizeClass}" placeholder="Placeholder text"></textarea>
<br/>
<textarea class="textarea${sizeClass}">Typed text</textarea>
<br/>
<textarea class="textarea${sizeClass}" disabled>Fixed text</textarea>`
}
